func shellSort<T: Comparable>(_ array: [T]) -> [T] {
    var a = array
    let n = a.count
    var gap = n / 2

    while gap > 0 {
        for i in gap..<n {
            let temp = a[i]
            var j = i
            while j >= gap && a[j - gap] > temp {
                a[j] = a[j - gap]
                j -= gap
            }
            a[j] = temp
        }
        gap /= 2
    }
    return a
}
